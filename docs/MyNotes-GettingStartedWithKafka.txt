	/projects/bin/kafka/bin/kafka-topics.sh --create --topic my_topic --zookeeper localhost:2181 --replication-factor 1 --partitions 1

	/projects/bin/kafka/bin/kafka-topics.sh --list  --zookeeper localhost:2181

- Creating a producer
	/projects/bin/kafka/bin/kafka-console-producer.sh --broker-list localhost:9092 --topic my_topic

- Creating the consumer
	/projects/bin/kafka/bin/kafka-console-consumer.sh --zookeeper localhost:2181 --topic my_topic --from-beginning
